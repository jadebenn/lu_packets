7/// Client-received general messages.
8#[derive(Debug, Deserialize, PartialEq, Serialize, VariantTests)]
9#[post_disc_padding = 1]
10#[repr(u32)]
11pub enum GeneralMessage {
12    Handshake(Handshake),
13    DisconnectNotify(DisconnectNotify),
14}
15
16/**
17    Completes a version handshake initiated by the client.
18
19    ### Trigger
20    Receipt of a client-sent [`Handshake`](super::server::Handshake) packet that was acceptable to the server.
21
22    ### Handling
23    Optionally check if the server's [`network_version`](Self::network_version) matches your own. You can usually assume that the server will check this itself and disconnect if it doesn't match, but you can check again to be sure.
24
25    ### Response
26    If the server's [`service_id`](Self::service_id) is [`ServiceId::Auth`], respond with a [`LoginRequest`](crate::auth::server::LoginRequest) with your username and password. If it is [`ServiceId::World`], send a [`ClientValidation`](crate::world::server::ClientValidation) with your username and the session key provided by auth.
27
28    ### Notes
29    As the version confirm process was designed with more than just client-server in mind, it sends the server's network version and service id as well, even though this isn't really needed by the client (even the service id isn't needed, since you usually only connect to auth once, and it's the very first connection). This could be simplified if the protocol is ever revised.
30*/
31#[derive(Debug, Deserialize, PartialEq, Serialize)]
32#[trailing_padding = 41]
33pub struct Handshake {
34    /// The network protocol version of the server. For servers compatible with live, this is `171022`. This was relevant mainly back when LU was actively updated. Server projects making modifications to the network protocol should set this to a different value.
35    pub network_version: u32,
36    /// Service ID of the server, [`ServiceId::Auth`] for auth servers, [`ServiceId::World`] for world servers.
37    #[padding = 4]
38    pub service_id: ServiceId,
39}
40
41/**
42    Notifies the client when it was actively disconnected by the server.
43
44    ### Trigger
45    Being disconnected by the server, the exact trigger depends on the variant.
46
47    ### Handling
48    Display a message to the user.
49
50    ### Response
51    None. Expect the connection to be closed shortly after, so a response won't even be possible.
52
53    ### Notes
54    You can be disconnected without receiving this packet, for example when your connection is lost. The server is also not obligated to send this packet and may disconnect you without doing so.
55*/
56#[derive(Debug, Deserialize, PartialEq, Serialize)]
57#[repr(u32)]
58pub enum DisconnectNotify {
59    /// Unspecified disconnect reason.
60    UnknownServerError,
61    /// The client's [`network_version`](super::server::Handshake::network_version) did not match the server's [`network_version`](Handshake::network_version). The message contains the server network version number.
62    WrongGameVersion(u32),
63    /// Unused for client-server.
64    WrongServerVersion(u32),
65    /// Connection attempt on invalid port, server emulators probably won't send this as they usually won't have server-server communication using LU's protocol.
66    ConnectionOnInvalidPort,
67    /// There was another login with your account and your session has been closed in favor of the new login.
68    DuplicateLogin,
69    /// The server is shutting down.
70    ServerShutdown,
71    /// No server hosting this map is available.
72    UnableToLoadMap,
73    /// The provided [`ClientValidation::session_key`](crate::world::server::ClientValidation::session_key) is incorrect.
74    InvalidSessionKey,
75    /// Server did not expect a [`ClientValidation`](crate::world::server::ClientValidation) at this time.
76    AccountNotInPendingList,
77    /// The provided [`CharacterLoginRequest::char_id`](crate::world::server::CharacterLoginRequest::char_id) was not a valid character ID of this account.
78    CharacterNotFound,
79    /// The character seems to be corrupted in the database.
80    CharacterCorruption,
81    /// You were kicked from the server.
82    Kick,
83    /// Error saving or loading progress.
84    SaveFailure,
85    /// The account's time-limited free trial expired, unused.
86    FreeTrialExpired,
87    /// The parental controls for this account prevent it from further play.
88    PlayScheduleTimeUp,
89}
