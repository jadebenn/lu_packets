7/**
8    Creates an [`Amf3::Array`](crate::world::amf3::Amf3::Array) containing the arguments.
9
10    Since AMF3 arrays are both vectors and maps at the same time, there are multiple forms of the macro, for map and for vector usage.
11
12    ### Map usage
13
14    The syntax is `name: value`, where `name` is a string literal that will be converted to an [`Amf3String`](crate::world::amf3::Amf3String), and `value` is an expression that will be converted to an [`Amf3`] object.
15
16    Example:
17
18    ```
19    # #[macro_use] extern crate lu_packets;
20    # use lu_packets::amf3;
21    # fn main() {
22    amf3! {
23        "false": false,
24        "true": true,
25        "double1": 3.14f32,
26        "double2": 3.14f64,
27        "string": "string",
28        "array": amf3! { "inner": "array"},
29    };
30    # }
31    ```
32
33    ### Vector usage
34
35    The syntax is the exact same as with the [`vec!`] macro, except that the arguments will be converted to an [`Amf3`] object before being inserted.
36
37    Example:
38
39    ```
40    # #[macro_use] extern crate lu_packets;
41    # use lu_packets::amf3;
42    # fn main() {
43    amf3! [true, false, true];
44    amf3! [true; 4];
45    # }
46    ```
47
48    [`Amf3`]: crate::world::amf3::Amf3
49*/
50#[macro_export]
51macro_rules! amf3 {
52	{} => { $crate::world::amf3::Amf3::Array($crate::world::amf3::Amf3Array::new()) };
53	($($name:literal:$value:expr),+ $(,)?) => {
54		{
55			let mut array = $crate::world::amf3::Amf3Array::new();
56			$(array.map.insert(::std::convert::TryInto::try_into($name).unwrap(), ::std::convert::TryInto::try_into($value).unwrap());)*
57			$crate::world::amf3::Amf3::Array(array)
58		}
59	};
60	($value:expr; $n:expr) => {
61		{
62			let converted = ::std::convert::TryInto::try_into($value).unwrap();
63			let mut array = $crate::world::amf3::Amf3Array {
64				map: ::std::collections::HashMap::new(),
65				vec: vec![converted; $n],
66			};
67			$crate::world::amf3::Amf3::Array(array)
68		}
69	};
70	($($value:expr),+ $(,)?) => {
71		{
72			let mut array = $crate::world::amf3::Amf3Array::new();
73			$(array.vec.push(::std::convert::TryInto::try_into($value).unwrap());)*
74			$crate::world::amf3::Amf3::Array(array)
75		}
76	};
77}

79/**
80    Creates a [`LuNameValue`] containing the arguments.
81
82    The syntax is `name: value`, where `name` is a string literal that will be converted to a [`LuVarWString<u32>`], and `value` is an expression that will be converted to an [`LnvValue`].
83
84    Example:
85
86    ```
87    # #[macro_use] extern crate lu_packets;
88    # use lu_packets::lnv;
89    # fn main() {
90    lnv! {
91        "wstring": "string expression",
92        "i32": 42i32,
93        "f32": 3.14f32,
94        "f64": 3.14f64,
95        "u32": 42u32,
96        "bool": true,
97        "i64": i64::MAX,
98        "u64": u64::MAX,
99        "string": b"byte slice",
100    };
101    # }
102    ```
103
104    Care should be taken with integer and float literals to suffix them with the correct type, as seen above. Rust assumes `i32` for integer and `f64` for float literals by default, which may not be what you want, and can lead to incorrect serialization.
105
106    [`LuNameValue`]: crate::world::LuNameValue
107    [`LuVarWString<u32>`]: crate::common::LuVarWString
108    [`LnvValue`]: crate::world::LnvValue
109*/
110#[macro_export]
111macro_rules! lnv {
112	{} => { $crate::world::LuNameValue::new() };
113	{$($name:literal:$value:expr,)*} => {
114		{
115			let mut lnv = $crate::world::LuNameValue::new();
116			$(lnv.insert(::std::convert::TryInto::try_into($name).unwrap(), $value.into());)*
117			lnv
118		}
119	}
120}

122/**
123    Converts the argument to a LU string.
124
125    This forwards to the [`TryInto`] implementation on the argument, which means the macro is flexible and works with both string and wstring types, and both fixed and variable types, depending on context. Generally, to convert to a string type, pass a byte slice, and for a wstring type, pass a `&str` or `String`.
126
127    [`TryInto`]: std::convert::TryInto
128*/
129#[macro_export]
130macro_rules! lu {
131    ($str_lit:expr) => {
132        ::std::convert::TryInto::try_into($str_lit).unwrap()
133    };
134}
