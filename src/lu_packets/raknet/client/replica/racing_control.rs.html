48impl<R: Read> Deserialize<LE, BEBitReader<R>> for RacingControlConstruction {
49    fn deserialize(reader: &mut BEBitReader<R>) -> Res<Self> {
50        let activity_user_infos = ReplicaD::deserialize(reader)?;
51        let expected_player_count = ReplicaD::deserialize(reader)?;
52        let pre_race_player_infos = if reader.read_bit()? {
53            let mut infos = vec![];
54            while reader.read_bit()? {
55                let info = ReplicaD::deserialize(reader)?;
56                infos.push(info);
57            }
58            Some(infos)
59        } else {
60            None
61        };
62        let post_race_player_infos = if reader.read_bit()? {
63            let mut infos = vec![];
64            while reader.read_bit()? {
65                let info = ReplicaD::deserialize(reader)?;
66                infos.push(info);
67            }
68            Some(infos)
69        } else {
70            None
71        };
72        let race_info = ReplicaD::deserialize(reader)?;
73        let during_race_player_infos = if reader.read_bit()? {
74            let mut infos = vec![];
75            while reader.read_bit()? {
76                let info = ReplicaD::deserialize(reader)?;
77                infos.push(info);
78            }
79            Some(infos)
80        } else {
81            None
82        };
83        Ok(Self {
84            activity_user_infos,
85            expected_player_count,
86            pre_race_player_infos,
87            post_race_player_infos,
88            race_info,
89            during_race_player_infos,
90        })
91    }
92}

94impl<W: Write> Serialize<LE, BEBitWriter<W>> for &RacingControlConstruction {
95    fn serialize(self, writer: &mut BEBitWriter<W>) -> Res<()> {
96        crate::raknet::client::replica::ReplicaS::serialize(&self.activity_user_infos, writer)?;
97        crate::raknet::client::replica::ReplicaS::serialize(&self.expected_player_count, writer)?;
98        writer.write_bit(self.pre_race_player_infos.is_some())?;
99        if let Some(infos) = &self.pre_race_player_infos {
100            for info in infos {
101                writer.write_bit(true)?;
102                crate::raknet::client::replica::ReplicaS::serialize(info, writer)?;
103            }
104            writer.write_bit(false)?;
105        }
106        writer.write_bit(self.post_race_player_infos.is_some())?;
107        if let Some(infos) = &self.post_race_player_infos {
108            for info in infos {
109                writer.write_bit(true)?;
110                crate::raknet::client::replica::ReplicaS::serialize(info, writer)?;
111            }
112            writer.write_bit(false)?;
113        }
114        crate::raknet::client::replica::ReplicaS::serialize(&self.race_info, writer)?;
115        writer.write_bit(self.during_race_player_infos.is_some())?;
116        if let Some(infos) = &self.during_race_player_infos {
117            for info in infos {
118                writer.write_bit(true)?;
119                crate::raknet::client::replica::ReplicaS::serialize(info, writer)?;
120            }
121            writer.write_bit(false)?;
122        }
123        Ok(())
124    }
125}
