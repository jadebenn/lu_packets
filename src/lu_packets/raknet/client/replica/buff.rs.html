28impl<R: Read> Deserialize<LE, BEBitReader<R>> for BuffInfo {
29    fn deserialize(reader: &mut BEBitReader<R>) -> Res<Self> {
30        let buff_id = LERead::read(reader)?;
31        let time_left = ReplicaD::deserialize(reader)?;
32        let cancel_on_death = reader.read_bit()?;
33        let cancel_on_zone = reader.read_bit()?;
34        let cancel_on_damaged = reader.read_bit()?;
35        let cancel_on_remove_buff = reader.read_bit()?;
36        let cancel_on_ui = reader.read_bit()?;
37        let cancel_on_logout = reader.read_bit()?;
38        let cancel_on_unequip = reader.read_bit()?;
39        let cancel_on_damage_absorb_ran_out = reader.read_bit()?;
40        let added_by_teammate = reader.read_bit()?;
41        let apply_on_teammates = reader.read_bit()?;
42        let added_by_teammate = if added_by_teammate {
43            Some(LERead::read(reader)?)
44        } else {
45            None
46        };
47        let ref_count = LERead::read(reader)?;
48        Ok(Self {
49            buff_id,
50            time_left,
51            cancel_on_death,
52            cancel_on_zone,
53            cancel_on_damaged,
54            cancel_on_remove_buff,
55            cancel_on_ui,
56            cancel_on_logout,
57            cancel_on_unequip,
58            cancel_on_damage_absorb_ran_out,
59            added_by_teammate,
60            apply_on_teammates,
61            ref_count,
62        })
63    }
64}
65impl<W: Write> Serialize<LE, BEBitWriter<W>> for &BuffInfo {
66    fn serialize(self, writer: &mut BEBitWriter<W>) -> Res<()> {
67        LEWrite::write(writer, self.buff_id)?;
68        crate::raknet::client::replica::ReplicaS::serialize(&self.time_left, writer)?;
69        writer.write_bit(self.cancel_on_death)?;
70        writer.write_bit(self.cancel_on_zone)?;
71        writer.write_bit(self.cancel_on_damaged)?;
72        writer.write_bit(self.cancel_on_remove_buff)?;
73        writer.write_bit(self.cancel_on_ui)?;
74        writer.write_bit(self.cancel_on_logout)?;
75        writer.write_bit(self.cancel_on_unequip)?;
76        writer.write_bit(self.cancel_on_damage_absorb_ran_out)?;
77        writer.write_bit(self.added_by_teammate.is_some())?;
78        writer.write_bit(self.apply_on_teammates)?;
79        if let Some(x) = self.added_by_teammate {
80            LEWrite::write(writer, x)?;
81        }
82        LEWrite::write(writer, self.ref_count)?;
83        Ok(())
84    }
85}
