51impl<R: Read> Deserialize<LE, BEBitReader<R>> for StatsInfo {
52    #[rustfmt::skip]
53    fn deserialize(reader: &mut BEBitReader<R>) -> Res<Self> {
54		let cur_health = LERead::read(reader)?;
55		let max_health = LERead::read(reader)?;
56		let cur_armor  = LERead::read(reader)?;
57		let max_armor  = LERead::read(reader)?;
58		let cur_imag   = LERead::read(reader)?;
59		let max_imag   = LERead::read(reader)?;
60		let damage_absorption_points = LERead::read(reader)?;
61		let immunity     = reader.read_bit()?;
62		let is_gm_immune = reader.read_bit()?;
63		let is_shielded  = reader.read_bit()?;
64		let actual_max_health = LERead::read(reader)?;
65		let actual_max_armor  = LERead::read(reader)?;
66		let actual_max_imag   = LERead::read(reader)?;
67		let factions          = LERead::read(reader)?;
68		let is_smashable = reader.read_bit()?;
69		let is_dead      = reader.read_bit()?;
70		let is_smashed   = reader.read_bit()?;
71		let smashable_info = if is_smashable {
72			Some(LERead::read(reader)?)
73		} else {
74			None
75		};
76		Ok(Self {
77				cur_health,
78				max_health,
79				cur_armor,
80				max_armor,
81				cur_imag,
82				max_imag,
83				damage_absorption_points,
84				immunity,
85				is_gm_immune,
86				is_shielded,
87				actual_max_health,
88				actual_max_armor,
89				actual_max_imag,
90				factions,
91				is_dead,
92				is_smashed,
93				smashable_info,
94		})
95	}
96}

98impl<W: Write> Serialize<LE, BEBitWriter<W>> for &StatsInfo {
99    fn serialize(self, writer: &mut BEBitWriter<W>) -> Res<()> {
100        LEWrite::write(writer, self.cur_health)?;
101        LEWrite::write(writer, self.max_health)?;
102        LEWrite::write(writer, self.cur_armor)?;
103        LEWrite::write(writer, self.max_armor)?;
104        LEWrite::write(writer, self.cur_imag)?;
105        LEWrite::write(writer, self.max_imag)?;
106        LEWrite::write(writer, self.damage_absorption_points)?;
107        writer.write_bit(self.immunity)?;
108        writer.write_bit(self.is_gm_immune)?;
109        writer.write_bit(self.is_shielded)?;
110        LEWrite::write(writer, self.actual_max_health)?;
111        LEWrite::write(writer, self.actual_max_armor)?;
112        LEWrite::write(writer, self.actual_max_imag)?;
113        LEWrite::write(writer, &self.factions)?;
114        writer.write_bit(self.smashable_info.is_some())?;
115        writer.write_bit(self.is_dead)?;
116        writer.write_bit(self.is_smashed)?;
117        if let Some(x) = &self.smashable_info {
118            LEWrite::write(writer, x)?;
119        }
120        Ok(())
121    }
122}
